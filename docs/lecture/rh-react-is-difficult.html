# React는 어렵다

# 무엇이 React를 어렵게 할까요?

React 훅 등장 이전에는 용도에 따라 함수 컴포넌트와 클래스 컴포넌트로 나눠 사용했습니다. 익숙한 함수와 달리 클래스 컴포넌트는 상대적으로 배워야 할 것이 많아 많은 사용자로 하여금 React를 어렵게 만들었습니다.

요인 설명
모던 JavaScript (ES6+) 새로운 JavaScript를 잘 알아야 React를 사용할 수 있습니다.
특히, 클래스 컴포넌트를 설계/관리하려면 OOP (opens new window)에 익숙해야 합니다.
라이프 사이클 메서드 라이프 사이클 메서드 (opens new window)은 React 클래스 컴포넌트에서만 사용 가능합니다.
컴포넌트의 생성/갱신/제거 각 시점에서 실행 가능한 방법을 학습해야 합니다.
재사용 로직 (패턴) Render Props (opens new window), HOC (opens new window)와 같은 고급(어려운) 개념을 활용해 로직을 재사용합니다.

# 모던 JavaScript (ES 6+)

class 문법은 이전 Prototype 기반의 JavaScript 객체 지향 프로그래밍 문법보다 직관적이지만, 웹 환경에서의 JavaScript는 함수 위주로 작성된 탓인지 class 프로그래밍을 어려워 하는 이들이 많습니다.

import React, { Component } from 'react'


class Counter extends Component {
  state = { count: this.props.initialCount }
  
  increment = () => {
    this.setState({ count: this.state.count + 1 })
  }
  
  render() {
    const { count } = this.state
    return (
      <button type="button" onClick={this.increment}>
        카운트({ count })
      </button>
    )
  }
}

export default Counter

# 라이프 사이클 메서드

가상(virtual) DOM을 구성/비교해 실제 DOM을 업데이트 하는 React 세계에서 라이프 사이클 메서드는 특정 시점에 필요한 처리를 하는 데 매우 유용하지만, class 문법과 함께 이러한 처리 과정 또한 어렵게 느껴지는 이들이 많아 React를 어렵게 합니다.

import React, { Component } from 'react'
import UserContext from '../user-context'


class ChatFeed extends Component {

  static contextType = UserContext
  
  state = { isScrolledToBottom: true }
  
  componentDidMount() {
    this.subscribeToFeed()
    this.setDocumentTitle()
    this.subscribeToOnlineStatus()
    this.subscribeToGeoLocation()
  }
  
  componentWillUnmount() {
    this.unsubscribeFromFeed()
    this.restoreDocumentTitle()
    this.unsubscribeFromOnlineStatus()
    this.unsubscribeFromGeoLocation()
  }
  
  componentDidUpdate(prevProps, prevState) {
    // 이전 props와 업데이트 된 props를 비교 처리
  }
  
  render() {
    // (초기|갱신) 렌더링
  }
}

# 재사용 로직 (패턴)

Context API, Render Props, HOC와 같은 고급(어려운) 개념을 활용해 로직을 재사용 하는 패턴은 React를 어렵게 만들었습니다.

import React from 'react'
import UserContext from '../user-context'


function ChatFeed({feedId}) {
  return (
    <UserContext.Consumer>
      {user => (
        // 피드 구독
        <SubscribeToFeed feedId={feedId}>
          {posts => (
            <>
              // 문서 제목 업데이트
              <DocumentTitle title={posts.length} />
              // 온라인 상태
              <OnlineStatus>
                {isOnline => (
                  // 위치 정보 업데이트
                  <GeoLocation>
                    {location => (
                      <>
                        <StickyScroller>
                          <Posts posts={posts} />
                        </StickyScroller>
                        <PostInput feedId={feedId} user={user} />
                      </>
                    )}
                  </GeoLocation>
                )}
              </OnlineStatus>
            </>
          )}
        </SubscribeToFeed>
      )}
    </UserContext.Consumer>
  )
}
