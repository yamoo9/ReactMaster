  • 폼 전송 핸들링
  • 폼 유효성 검사
  • 오류 메시지 표시

# 폼 유효성 검사

formik의 유효성 검사(validation) (opens new window) 방식에 따라 사용자가 입력한 값을 검증하여 오류를 처리할 수 있습니다.

// 사용자 정의 유효성 검사 함수
const validate = values => {
  // 오류 수집 객체 정의
  const errors = {};

  if (!values.name.trim()) {
    errors.name = '`이름` 입력은 필수입니다.';
  } 
  else if (values.name.length > 1) {
    errors.name = '`이름`은 최소 2자 이상 입력되어야 합니다.';
  }

  if (!values.email.trim()) {
    errors.email = '`이메일` 입력은 필수입니다.';
  } 
  else if (!/^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,4}$/i.test(values.email)) {
    errors.email = '이메일 입력 형식이 올바르지 않습니다. 확인 후, 다시 입력하세요.';
  }

  // 오류 수집 객체 반환
  return errors;
};

# 오류 메시지 표시

formik의 오류(errors) 상태는 유효성 검사 유틸리티 함수에 의해 업데이트 되므로 다음과 같이 오류 상태 조건에 따라 화면에 메시지가 표시되도록 설정합니다.










 


<div className="formControl">
  <label htmlFor="userName">이름</label>
  <input
    type="text"
    id="userName"
    name="name"
    value={formik.values.name}
    onChange={formik.handleChange}
  />
  {formik.errors.name && <span role="alert">{formik.errors.name}</span>}
</div>

# 폼 UX 개선

사용자가 입력 중에 오류 메시지가 표시되면 압박을 받게 되어 안좋은 경험을 제시할 수 있습니다. 그러므로 사용자가 한 번이라도 입력한 후에 오류가 발생했을 때 메시지를 표시하는 것이 보다 사용자 경험을 좋게합니다.











 
 
 


<div className="formControl">
  <label htmlFor="userEmail">이메일</label>
  <input
    type="email"
    id="userEmail"
    name="email"
    value={formik.values.email}
    onChange={formik.handleChange}
    onBlur={formik.handleBlur}
  />
  {formik.touched.email && formik.errors.email && (
    <span role="alert">{formik.errors.email}</span>
  )}
</div>

# 폼 유효성 검사 스키마(설계)

Yup (opens new window)은 값 구문 분석 및 유효성 검사를 위한 JavaScript 스키마 빌더입니다. 스키마를 정의하고, 일치시킬 값을 변환하고, 기존 값의 모양을 검증할 수 있습니다. Yup 스키마는 매우 표현력이 뛰어나며 복잡하고 상호 의존적인 검증 또는 값 변환을 모델링 할 수 있습니다.

# Yup 패키지 설치

npm i yup
1

# Yup 패키지 불러오기

import * as Yup from 'yup'

Yup 라이브러리 사용법 (opens new window)을 참고해 유효성 검사 스키마를 작성합니다.

const validationSchema = Yup.object({
  
  name: Yup.string()
    .required('이름` 입력은 필수입니다.')
    .min(2, '`이름`은 최소 2자 이상 입력되어야 합니다.')
    .trim(), // 사용자가 입력한 양쪽 공백을 제거 (blur 이벤트 발생 시, 검사할 때마다 처리)
  
  email: Yup.string()
    .required('`이메일` 입력은 필수입니다.')
    .email('이메일 입력 형식이 올바르지 않습니다. 확인 후, 다시 입력하세요.'),
  
  channel: Yup.string()
    .required('채널` 입력은 필수입니다.')
    .min(2, '`채널`은 최소 2자 이상 입력되어야 합니다.')
    .trim(),

})

# getFieldProps를 활용한 리팩터링

formik은 폼 컨트롤을 위해 설정한 속성들(value, onChange, onBlur 등)을 개별 입력하지 않고도 제어할 수 있는 getFieldProps (opens new window) 메서드를 제공합니다. 이 메서드를 사용해 코드를 보다 간결하게 관리할 수 있습니다.







 
 
 
 






<div className="formControl">
  <label htmlFor="userName">이름</label>
  <input
    type="text"
    id="userName"
    name="name"
    // value={formik.values.name}
    // onChange={formik.handleChange}
    // onBlur={formik.handleBlur}
    {...formik.getFieldProps('name')}
  />
  {formik.touched.name && formik.errors.name && (
    <span role="alert">{formik.errors.name}</span>
  )}
</div>
