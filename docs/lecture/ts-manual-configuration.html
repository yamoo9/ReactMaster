# 매뉴얼 구성

TypeScript를 사용하기 위한 React 개발 환경 구성을 위해 디렉토리를 생성/이동한 후, 초기화 및 의존 패키지를 설치합니다.

# 프로젝트 디렉토리 생성/이동
md react-ts && cd $_

# Git, NPM 초기화
git init && npm init

# 의존 패키지 설치
npm i -D webpack webpack-cli webpack-dev-server \ 
         react react-dom \
         typescript @types/react @types/react-dom ts-loader

프로젝트 루트 위치에 webpack.config.js, tsconfig.json 파일을 작성합니다.

webpack.config.js

const path = require('path')

module.exports = {
  target: 'web',
  mode: 'production',
  devtool: 'eval',
  entry: {
    main: './src/index.tsx'
  },
  output: {
    path: path.join(__dirname, './public'),
    filename: 'js/[name].js'
  },
  module: {
    rules: [
      { 
        test: /\.tsx?$/i, 
        exclude: /node_modules/, 
        use: 'ts-loader' 
      }
    ]
  },
  resolve: {
    extensions: ['.ts', '.tsx', '.js', '.json']
  },
  devServer: {
    contentBase: path.join(__dirname, './public'),
    port: 3000,
    index: 'index.html',
    compress: true,
    hot: true
  }
}
