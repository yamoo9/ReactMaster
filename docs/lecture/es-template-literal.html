# 템플릿 리터럴

템플릿 리터럴(Template Literals) (opens new window)은 특별한 힘을 가진 문자(String) 입니다.

# ES6

let brand = '이듬(E.UID)'
let framework = 'React'

const course_name = `${brand}'s ${framework}`
console.log(course_name) // "이듬's React"
